Formato SQL de Entrada 

CREATE TABLE ventas (
    id INT PRIMARY KEY,
    producto VARCHAR(100),
    cantidad INT,
    precio DECIMAL(10, 2),
    fecha_venta DATE
);

Columna Tipo Descripción
id INT Identificador único de la venta
producto VARCHAR(100) Nombre del producto vendido
cantidad INT Cantidad de unidades vendidas
precio DECIMAL(10, 2) Precio unitario del producto
fecha_venta DATE Fecha en que se realizó la venta
